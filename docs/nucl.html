 16        """Create a new nucl_acid.
 17
 18        Args:
 19            sequence (Seq): the nucleic acid sequence.
 20            no_mod (list): A list() of 0 and 1 (ints), where 1 denotes nucleotides that CANNOT be modified (neither indels nor substitutions).
 21            no_indel (list): A list() of 0 and 1 (ints), where 1 denotes nucleotides that CANNOT have indel mutations adjacent to them.
 22            score_region (list): A list() of 0 and 1 (ints), where 1 denotes nucleotides that will be scored for high pair probability at temp - offset and low pair probability at temp + offset.
 23            design_parameters (design_parameters): the design parameters.
 24            is_rna (bool): whether the nucleic acid is DNA or RNA.
 25
 26        Raises:
 27            Exception: _description_
 28            Exception: _description_
 29            Exception: _description_
 30            ValueError: _description_
 31            ValueError: _description_
 32        """
 33        if len(no_mod) != len(sequence):
 34            raise Exception("no_mod length is not equal to seq length")
 35        if len(no_indel) != len(sequence):
 36            raise Exception("no_indel length is not equal to seq length")
 37        if len(score_region) != len(sequence):
 38            raise Exception("score_region length is not equal to seq length")
 39        self.sequence = sequence
 40        self.no_mod = no_mod
 41        self.no_indel = no_indel
 42        self.score_region = score_region
 43        self.is_rna = is_rna
 44
 45        #Check that definded RNAs do not contain DNA bases and vice versa
 46        if self.is_rna and self.sequence.find('T') != -1:
 47            raise ValueError
 48        elif (not self.is_rna) and self.sequence.find('U') != -1:
 49            raise ValueError
 50
 51        if self.sequence.find('N') != -1:
 52            self.sequence = MutableSeq(self.sequence)
 53            if self.is_rna:
 54                nucleotides = ['A', 'U', 'G', 'C']
 55            else:
 56                nucleotides = ['A', 'T', 'G', 'C']
 57            while self.sequence.find('N') != -1:
 58                self.sequence[self.sequence.find('N')] = nucleotides[random.getrandbits(2)]
 59            self.sequence = Seq(self.sequence)
 60        #Score the entire thing if everything was set to 0. Have to score something at minimum
 61        if score_region.count(0) == len(score_region):
 62            self.score_region = [1] * len(score_region)
 63
 64        self.fitness_score(design_parameters)

187    def save(self, path:str) -> None:
188        """Write the nucl_set to an SPSS formatted .fastq file.
189
190        Args:
191            path (str): Path to save the file. If no directory specified will save to source directory.
192        """
193        #raises IOError if path is invalid, overwrites existing files
194        #This saves to a .fastq, where the quality scores are used instead for bitwise indicating nomod, noindel, scoreregion
195        #This will be done with ASCII 0-7, p-w  [XXX011Y]
196        #These were chosen to avoid any extraneous ASCII characters.
197        #b1: no_mod
198        #b2: no_indel
199        #b3: score_region
200        #b7: DNA / RNA
201        # AS SANGER SCORES:
202        # 0: 15, 7: 22
203        # p: 79, w: 86
204        with open(path, 'w') as handle:
205            for i in range(0, len(self)):
206                nucl = self.nucls[i]
207                if nucl.is_rna:
208                    offset = 79
209                else:
210                    offset = 15
211                quals = list()
212                for j in range (0, len(nucl)):
213                    bitString = str(nucl.no_mod[j]) + str(nucl.no_indel[j]) + str(nucl.score_region[j])
214                    bitsAsInt = int(bitString, 2)
215                    quals.append(bitsAsInt + offset)
216                record = SeqRecord(seq=self.nucls[i].sequence, id=str(i), description="score=" + str(self.nucls[i].score))
217                record.letter_annotations["phred_quality"] = quals
218                SeqIO.write(record, handle=handle, format='fastq')
219                del quals
220                del record
221                del nucl

223    def read(self, path:str, design_parameters:design_parameters) -> None:
224        """Reads an SPSS formatted .fastq file and appends its nucl_acid's to the nucl_set.
225
226        Args:
227            path (str): Path to read the file. If no directory specified will assume source directory.
228            design_parameters (design_parameters): the design parameters.
229
230        Raises:
231            ValueError: _description_
232            ValueError: _description_
233            ValueError: _description_
234            ValueError: _description_
235            ValueError: _description_
236            ValueError: _description_
237        """
238        #If any other characters are encountered, an error should be raised.
239        for record in SeqIO.parse(path, "fastq"):
240            quals = record.letter_annotations["phred_quality"]
241            if max(quals) > 86 or min(quals) < 15:
242                raise ValueError("NOT SPSS")
243            elif max(quals) > 22: # Either RNA or invalid
244                if max(quals) < 79:
245                    raise ValueError("NOT SPSS")
246                if min(quals) < 79:
247                    raise ValueError("NOT SPSS")
248                offset = 79
249                is_rna = True
250            elif min(quals) <  79: # Either DNA or invalid
251                if min(quals) > 22:
252                    raise ValueError("NOT SPSS")
253                if max(quals) > 22:
254                    raise ValueError("NOT SPSS")
255                offset = 15
256                is_rna = False
257            else:
258                raise ValueError("NOT SPSS")
259            no_mod = list()
260            no_indel = list()
261            score_region = list()
262            for qual in quals:
263                #Convert qual score to int. Backfill 0s to make 3 bit int
264                bits_string = str(bin(qual - offset))[2:].zfill(3)
265                no_mod.append(int(bits_string[0] == '1'))
266                no_indel.append(int(bits_string[1] == '1'))
267                score_region.append(int(bits_string[2] == '1'))
268            self.append(new_nucl_acid=nucl_acid(sequence=record.seq, no_mod=no_mod, no_indel=no_indel, score_region=score_region, is_rna=is_rna, design_parameters=design_parameters))

270def mutate(nucl:nucl_acid, design_parameters:design_parameters) -> nucl_acid:
271    """Mutates a nucl_acid given design parameters and returns a mutated nucl_acid. Does not modify the original.
272
273    Args:
274        nucl (nucl_acid): the nucl_acid to make a mutant of.
275        design_parameters (design_parameters): the design parameters.
276
277    Raises:
278        ValueError: _description_
279
280    Returns:
281        nucl_acid: a new nucl_acid.
282    """
283    #I'm trying to make this function as fast as possible since it will be called once per every single
284    #nucleotide in a sequence to generate one variant.
285    #For larger pool sizes this can really stack up.
286    
287    if len(nucl.no_mod) != len(nucl.sequence) or len(nucl.no_indel) != len(nucl.score_region) or len(nucl.no_mod) != len(nucl.no_indel):
288        raise ValueError
289
290    #As of writing the design parameters function forces the weights to be whole numbers between 0 and 16.
291    #This could be made a touch faster explicitly using ints.
292    weights = copy.copy(design_parameters.weights)
293    weights_total = sum(weights)
294
295    #Make the weights array into an array of increasing values which can be compared to a random int
296    for i in range(0, len(weights)):
297        weights[i] = weights[i - 1] + weights[i]
298
299    sequence = list(copy.copy(str(nucl.sequence)))
300    no_mod = copy.copy(nucl.no_mod)
301    no_indel = copy.copy(nucl.no_indel)
302    score_region = copy.copy(nucl.score_region)
303
304    #Define nucleotide set
305    #Really don't want to be repeatedly checking if RNA or DNA inside loop
306    if nucl.is_rna:
307        nucleotides = ['A', 'U', 'G', 'C']
308    else:
309        nucleotides = ['A', 'T', 'G', 'C']
310
311    #mod_range = list(range(0, len(sequence)))
312
313    i = 0
314    while i < len(sequence):
315        if no_mod[i]:
316            i = i+1
317            continue
318        choice = random.randint(0, weights_total)
319
320        #Comparing the generated int with values in the array of increasing ints.
321        selection = 6
322        for j in range(0, len(weights)):
323            if choice > weights[j]:
324                continue
325            selection = j
326            break
327        
328        #If no change selected, move on
329        if selection == 6:
330            i = i+1
331            continue
332        
333        #If indels not allowed for this position, continue
334        if (selection == 4 or selection == 5) and no_indel[i]:
335            i = i+1
336            continue
337        
338        #Index the nucleotide array instead of checking several times. Hope it's faster
339        #It is fewer lines, anyway
340        if selection >= 0 and selection < 4:
341            sequence[i] = nucleotides[selection]
342            i = i+1
343            continue
344
345        # Deletion is faster than insertion so I've moved it here for now
346        if selection == 5:
347            del sequence[i]
348
349            #The other arrays need to be the same size!
350            del no_mod[i]
351            del no_indel[i]
352            del score_region[i]
353            #If this was the last nucleotide, the while loop will see that anyway
354            #Repeat this nucleotide for the next mod so no incrementing i
355            continue
356
357        # If an insertion was selected
358        #Unnecessary check (selection == 5) but I'll leave it here in case the order changes
359        #TODO: remove unnecessary check
360        if selection == 4:
361            #Generate a random nucleotide from nucleotide list (getrandbits (2) will generate an int 0-4)
362            new_nt = nucleotides[random.getrandbits(2)]
363            
364
365            #I'm pretty worried about the score region changing code. It could be slow and could have bug(s)
366            
367            #Randomly generate bit to determine whether to insert to the left or right
368            #If bit was 1, insert to the left
369            if random.getrandbits(1):
370                sequence.insert(i, new_nt)
371                no_mod.insert(i, 0)
372                no_indel.insert(i, 0)
373
374
375                #If at position 0, there is nothing to the left!
376                if i == 0 and score_region[0]:
377                    score_region.insert(i, 1)
378                    i = i+2
379                    continue
380                #At any other position besides 0, check to the left and right
381                elif score_region[i - 1] and score_region[i]:
382                    score_region.insert(i, 1)
383                    i = i+2
384                    continue
385                
386                #default behavior is to insert a 0
387                score_region.insert(i, 0)
388                i = i+2
389                continue
390            #Else, insert to the right
391
392            #Check if we are at the end - use append instead
393            if i == len(sequence) - 1:
394                sequence.append(new_nt)
395                no_mod.append(0)
396                no_indel.append(0)
397
398                #If we are at the end and it's scored, extend the score region
399                if score_region[i]:
400                    score_region.append(1)
401                #Othwerise, if we are at the end and it isn't scored, don't extend the score region
402                else:
403                    score_region.append(0)
404                #Exit the loop because that's it
405                break
406            
407            #Otherwise, we insert
408            sequence.insert(i+1, new_nt)
409            no_mod.insert(i+1, 0)
410            no_indel.insert(i+1, 0)
411            
412            if score_region[i] and score_region[i+1]:
413                score_region.insert(i+1, 1)
414                i = i+2
415                continue
416            
417            #default behavior is to insert a 0
418            score_region.insert(i+1, 0)
419            i = i+2
420            #Unnecessary but I'll leave it here in case the order changes
421            #TODO: remove unnecessary continue
422            continue
423    
424    return nucl_acid(sequence=Seq(''.join(sequence)), no_mod=no_mod, no_indel=no_indel, score_region=score_region,
425                     design_parameters=design_parameters, is_rna=nucl.is_rna)

272    """Mutates a nucl_acid given design parameters and returns a mutated nucl_acid. Does not modify the original.
273

275        nucl (nucl_acid): the nucl_acid to make a mutant of.
276        design_parameters (design_parameters): the design parameters.

278    Raises:
279        ValueError: _description_
280
281    Returns:
282        nucl_acid: a new nucl_acid.
283    """
284    #I'm trying to make this function as fast as possible since it will be called once per every single
285    #nucleotide in a sequence to generate one variant.
286    #For larger pool sizes this can really stack up.
287    
288    if len(nucl.no_mod) != len(nucl.sequence) or len(nucl.no_indel) != len(nucl.score_region) or len(nucl.no_mod) != len(nucl.no_indel):
289        raise ValueError

291    #As of writing the design parameters function forces the weights to be whole numbers between 0 and 16.
292    #This could be made a touch faster explicitly using ints.
293    weights = copy.copy(design_parameters.weights)
294    weights_total = sum(weights)
295
296    #Make the weights array into an array of increasing values which can be compared to a random int
297    for i in range(0, len(weights)):
298        weights[i] = weights[i - 1] + weights[i]
299
300    sequence = list(copy.copy(str(nucl.sequence)))
301    no_mod = copy.copy(nucl.no_mod)
302    no_indel = copy.copy(nucl.no_indel)
303    score_region = copy.copy(nucl.score_region)
304
305    #Define nucleotide set
306    #Really don't want to be repeatedly checking if RNA or DNA inside loop
307    if nucl.is_rna:
308        nucleotides = ['A', 'U', 'G', 'C']
309    else:
310        nucleotides = ['A', 'T', 'G', 'C']

312    #mod_range = list(range(0, len(sequence)))
313
314    i = 0
315    while i < len(sequence):
316        if no_mod[i]:
317            i = i+1
318            continue
319        choice = random.randint(0, weights_total)
320
321        #Comparing the generated int with values in the array of increasing ints.
322        selection = 6
323        for j in range(0, len(weights)):
324            if choice > weights[j]:
325                continue
326            selection = j
327            break
328        
329        #If no change selected, move on
330        if selection == 6:
331            i = i+1
332            continue
333        
334        #If indels not allowed for this position, continue
335        if (selection == 4 or selection == 5) and no_indel[i]:
336            i = i+1

338

340        #It is fewer lines, anyway
341        if selection >= 0 and selection < 4:

347        if selection == 5:

402                #Othwerise, if we are at the end and it isn't scored, don't extend the score region
403                else:
